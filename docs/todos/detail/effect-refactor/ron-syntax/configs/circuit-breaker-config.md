# Name
CircuitBreakerConfig

# Parameters
- bumps_required: u32
- spawn_count: u32
- inherit: bool
- shockwave_range: f32
- shockwave_speed: f32

# Description
- bumps_required: Bumps needed per cycle before the reward fires
- spawn_count: Number of extra bolts spawned as the reward
- inherit: Whether spawned bolts inherit effect trees
- shockwave_range: Maximum radius of the reward shockwave
- shockwave_speed: Expansion speed of the reward shockwave
